%{
#include <stdio.h>
#include <stdlib.h>
void yyerror(const char *s){ fprintf(stderr,"Parse error: %s\n", s); }
int yylex(void);
%}
%union { int ival; }
%token <ival> NUMBER

%left '+' '-'
%left '*' '/'
%right UMINUS

%%
input: /* empty */
     | input line
     ;
line: expr '\n' { printf("Result = %d\n", $1); }
    | '\n'
    ;
expr: NUMBER               { $$ = $1; }
    | expr '+' expr       { $$ = $1 + $3; }
    | expr '-' expr       { $$ = $1 - $3; }
    | expr '*' expr       { $$ = $1 * $3; }
    | expr '/' expr       { $$ = $1 / $3; }
    | '-' expr %prec UMINUS { $$ = -$2; }
    | '(' expr ')'        { $$ = $2; }
    ;
%%
int main(void){ printf("Enter infix expr (end with newline)\\n"); yyparse(); return 0; }
